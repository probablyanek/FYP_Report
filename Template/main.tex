\chapter*{Codes and Standards}

\noindent\textit{Please refer to the following example.}

\noindent\textbf{Project Title:} \textit{Flood Monitoring and Alert System}

\begin{enumerate}

  \item \textbf{Environmental \& Hydrological Standards}
  \begin{itemize}
    \item \textbf{ISO~14001 -- Environmental Management Systems}\\
          Ensures sustainable monitoring and minimal environmental impact.
    \item \textbf{IS~4987 (India) -- Rain Gauge Specifications}\\
          Used for accurate rainfall data collection.
  \end{itemize}

  \item \textbf{Sensor \& Measurement Standards}
  \begin{itemize}
    \item \textbf{IEC~60529 -- IP Rating (Ingress Protection)}\\
          Ensures water level sensors and enclosures are waterproof (e.g., IP67/IP68).
    \item \textbf{IEEE~1451 -- Smart Transducer Interface}\\
          Standard for integrating sensors with microcontrollers.
  \end{itemize}

  \item \textbf{Communication \& IoT Standards}
  \begin{itemize}
    \item \textbf{IEEE~802.15.4 -- Low-Rate Wireless Personal Area Networks}\\
          Used in Zigbee-based sensor networks.
    \item \textbf{IEEE~802.11 -- Wi-Fi Communication}\\
          For transmitting data to cloud/server.
    \item \textbf{LoRaWAN Specification (v1.x)}\\
          Long-range, low-power communication for remote monitoring.
    \item \textbf{MQTT Protocol (ISO/IEC~20922)}\\
          Lightweight messaging protocol for IoT data transmission.
  \end{itemize}

  \item \textbf{Embedded Systems \& Electrical Safety}
  \begin{itemize}
    \item \textbf{IEC~60364 -- Electrical Installations Safety}\\
          Ensures safe deployment of electronics in field conditions.
    \item \textbf{ISO/IEC~27001 -- Information Security}\\
          For secure transmission of sensor data and alerts.
  \end{itemize}

  \item \textbf{Power \& Energy Standards (for field deployment)}
  \begin{itemize}
    \item \textbf{IEC~62133 -- Battery Safety}\\
          For rechargeable batteries in remote sensor nodes.
  \end{itemize}

\end{enumerate}
